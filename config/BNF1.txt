<BOOL'> ::= "#" <BOOL> "#";
<BOOL> ::= <BOOL> "||" <JOIN>
            | <JOIN>;
<JOIN> ::= <JOIN> "&&" <NOT>
            | <NOT>;
<NOT> ::= <REL> 
            | "!" <REL>;
<REL> ::= <EXPR> ">" <EXPR> 
            | <EXPR> ">=" <EXPR> 
            | <EXPR> "<" <EXPR> 
            | <EXPR> "<=" <EXPR> 
            | <EXPR> "==" <EXPR> 
            | <EXPR> "!=" <EXPR> 
            | BCONST;
<EXPR> ::= <EXPR> "+" <TERM>
            | <EXPR> "-" <TERM>
            | <TERM>;
<TERM> ::= <TERM> "*" <NEGA>
            | <TERM> "/" <NEGA>
            | <NEGA>;
<NEGA> ::= <FACTOR>
            | M <FACTOR>;
<FACTOR> ::= "(" <EXPR> ")"
            | ID
            | NUM;
